const gerarPix = async () => {
    setLoadingPix(true);
    setQrCodeBase64(null);
    setQrString(null);
    setPaymentId(null);
    setAguardandoPix(false);
    console.log("🟢 gerarPix SITE chamado");

    try {
      const token = localStorage.getItem("token");
      const usuario = localStorage.getItem("usuario");
      const nomeCliente = usuario ? JSON.parse(usuario).nome : "Cliente";

      console.log("pedidoId 👉", pedidoId);
      console.log("totalFinal 👉", totalFinal);

      const gerarPixPDV = async () => {
        console.log("🔵 gerarPixPDV chamado");

        try {
          setLoadingPix(true);

          const token = localStorage.getItem("token");
          const usuario = localStorage.getItem("usuario");
          const nomeCliente = usuario
            ? JSON.parse(usuario).nome
            : "Cliente PDV";

          // 👉 cria pedido + pagamento PIX no CAIXA
          const res = await axios.post(
            `${API_URL}/caixa/finalizar`,
            {
              clienteId,
              itens, // vem do CaixaLoja
              metodoPagamento: "PIX",
              valorTotal: total,
            },
            {
              headers: {
                Authorization: token ? `Bearer ${token}` : "",
              },
            },
          );

          console.log("Resposta PDV PIX:", res.data);

          const { pagamento } = res.data || {};

          setQrCodeBase64(pagamento?.pixQrCodeBase64 ?? null);
          setQrString(pagamento?.pixCodigo ?? null);
          setPaymentId(pagamento?.pixTxid ?? null);

          setAguardandoPix(true);
        } catch (err) {
          console.error("Erro PDV PIX:", err);
          toast.error("Erro ao gerar PIX do PDV");
        } finally {
          setLoadingPix(false);
        }
      };

    } catch (err) {
      console.error(err);
      toast.error("Erro ao gerar PIX");
    } finally {
      setLoadingPix(false);
    }
  };

  useEffect(() => {
    if (!paymentId) return;

    const interval = setInterval(async () => {
      try {
        const res = await axios.get(
          // `${API_URL}/pedidos/${pedidoId}`
          `${API_URL}/pagamentos/status/${paymentId}`,
        );

        if (res.data?.status === "paid" || res.data?.status === "approved") {
          clearInterval(interval);
          toast.success("PIX confirmado!");
          await onConfirm({
            metodoPagamento: "PIX",
            totalComJuros: Number(totalFinal),
          });
          onClose();
        }
      } catch (err) {
        console.log("Aguardando PIX...");
      }
    }, 3000);

    return () => clearInterval(interval);
  }, [paymentId, totalFinal, onConfirm, onClose]);

  const confirmarPagamento = async () => {
    if (!forma) {
      toast.info("Selecione a forma de pagamento");
      return;
    }

    // SITE → Crédito = Stripe
    if (isSite && forma === "CREDITO") {
      await onConfirm({
        metodoPagamento: "CREDITO",
        parcelas,
        totalComJuros: Number(totalFinal),
      });
      return;
    }

    if (isSite && forma === "PIX") {
      toast.info("Aguardando pagamento PIX");
      return;
    }

    const troco =
      forma === "DINHEIRO"
        ? Math.max(Number(valorPago || 0) - Number(totalFinal), 0)
        : 0;

    await onConfirm({
      metodoPagamento: forma,
      parcelas,
      totalComJuros: Number(totalFinal),
      troco,
    });

    onClose();
  };
